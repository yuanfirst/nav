# Cloudflare Pages 环境变量配置模板
# 
# 注意：此文件仅作为参考模板，实际环境变量需要在 Cloudflare Dashboard 中配置
# 配置位置：Settings > Variables and Secrets
#
# 重要说明：
# - 首次部署时的"环境变量 (高级)"选项不会保存环境变量
# - 必须在 Cloudflare Dashboard 的 Settings 页面手动添加
# - 环境变量可以在部署前或部署后配置

# ============================================
# 必须的环境变量（不配置将无法使用管理功能）
# ============================================

# 管理员用户名
ADMIN_USERNAME=admin

# 管理员密码
ADMIN_PASSWORD=your_password_here

# JWT 密钥，用于生成登录 token
# 生成方式：openssl rand -base64 32
# 或使用在线工具生成至少32位的随机字符串
JWT_SECRET=your_jwt_secret_here_at_least_32_characters

# ============================================
# 可选的环境变量（不配置不影响基本功能）
# ============================================

# OpenAI API Key（AI 功能）
# 如果不配置，可以在设置界面中配置（登录后进入"设置" → "AI 助手"）
OPENAI_API_KEY=sk-...

# OpenAI API 基础地址（可选）
OPENAI_BASE_URL=https://api.openai.com/v1

# OpenAI 模型名称（可选）
OPENAI_MODEL=gpt-4o-mini

# OpenAI 认证 Header 名称（可选）
OPENAI_AUTH_HEADER=Authorization

# OpenAI 认证前缀（可选）
OPENAI_AUTH_PREFIX=Bearer
